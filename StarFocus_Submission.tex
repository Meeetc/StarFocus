\section{Problem Statement}
% ============================================================

\begin{quote}
\textit{``University students lose an estimated 2--4 productive hours daily to decision fatigue, task paralysis, and dopamine-driven attention drift. There exists no unified platform that (a) intelligently ranks academic tasks by urgency and impact, (b) applies behavioral nudges at the point of distraction, and (c) harnesses social accountability to sustain deep work --- all within the student's existing academic ecosystem.''}
\end{quote}

\subsection{Problem Description}

\begin{enumerate}[leftmargin=*, label=\textbf{\arabic*.}]
    \item \textbf{Decision Fatigue:} Students spend 30--45 minutes daily just figuring out \textit{what} to study. With 5--7 courses, each with overlapping deadlines and varying grade weightages, the cognitive cost of choosing is itself a productivity killer.

    \item \textbf{Task Paralysis:} Even after choosing a task, students feel overwhelmed by its perceived size. Without clear guidance on where to begin, procrastination becomes the default response.

    \item \textbf{Attention Fragmentation:} The average student picks up their phone 80+ times per day. Each context switch costs 23 minutes of refocus time (Mark et al., 2005), turning a 4-hour study block into barely 90 minutes of actual deep work.

    \item \textbf{Lack of Motivation:} Studying is an invisible activity --- there is no social validation, no visible progress, and no accountability. Unlike fitness (Strava) or social media (likes), focus effort goes entirely unrecognized.

    \item \textbf{Tool Fragmentation:} Students juggle Google Calendar, Notion, Forest, and Todoist --- none of which talk to each other or to the university's LMS. The result is more overhead, not less.
\end{enumerate}

% ============================================================
\section{Theme: Attention Economy on Campus}
% ============================================================

\textbf{Selected Theme:} \textit{Theme A --- Attention Economy on Campus}

\begin{quote}
\textit{``College isn't short on time. It's short on attention.''}
\end{quote}

\subsection{Justification of Theme Relevance}

\begin{enumerate}[leftmargin=*, label=\textbf{\arabic*.}]
    \item \textbf{Attention is the core bottleneck, not time:} Students have 6--8 hours of free time daily, yet productive output averages only 1.5--2 hours. StarFocus directly targets this attention gap by reducing decision fatigue (intelligent ranking) and protecting focus time (workload-adaptive interventions).

    \item \textbf{Notifications, reels, and social pressure are the enemy:} The theme identifies exactly what StarFocus combats --- dopamine-driven digital distraction. Our three-layer intervention system (breathing exercise $\rightarrow$ greyscale $\rightarrow$ vibration) applies friction precisely at the moment of distraction, not through blanket blocking.

    \item \textbf{Helping students take control of their time:} StarFocus's priority ranking engine with Google Classroom integration eliminates the ``what should I work on?'' problem entirely, giving students an auto-prioritized, intelligent dashboard --- no manual setup needed.

    \item \textbf{Reducing mental clutter:} By aggregating all obligations (academic via Google Classroom + clubs/extracurriculars via manual entry) into a single workload score (0--100), StarFocus replaces scattered to-do lists with one clear number that tells students exactly how much pressure they are under.

    \item \textbf{Making better decisions about where focus goes:} The social motivation layer (leaderboards, study groups, Strava-style progress charts) makes focus \textit{visible and rewarding} --- transforming an invisible activity into a recognized, competitive, and habit-forming experience.
\end{enumerate}

\section{Technical Approach}
% ============================================================

StarFocus is a \textbf{three-phase Android application} built around three psychological primitives:

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabularx}{\textwidth}{|l|l|X|}
\hline
\textbf{Phase} & \textbf{Name} & \textbf{Purpose} \\
\hline
Phase 1 & \textbf{The Brain} & Intelligent task ranking via Google Classroom API sync + manual entry. Priority formula: $\dfrac{\text{Grade Weightage} \times (1 - \text{Completion\%})}{\text{Time Remaining}} \times \text{Quiz Prep Multiplier}$ \\
\hline
Phase 2 & \textbf{The Shield} & Workload-adaptive distraction intervention: breathing exercise $\rightarrow$ greyscale mode $\rightarrow$ continuous vibration. Timers scale with workload score (Low / Moderate / High). \\
\hline
Phase 3 & \textbf{The Hook} & Social motivation: opt-in global leaderboard, study groups, Strava-style progress charts, streaks with freeze tokens, and badge system. \\
\hline
\end{tabularx}
\end{center}
